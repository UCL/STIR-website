131 template <class A>
132 shared_ptr<VoxelsOnCartesianGrid<float> > IOTests<A>::create_single_image()
133 {
134  // construct density image
135 #ifdef HAVE_LLN_MATRIX
136  USING_NAMESPACE_ECAT
137  USING_NAMESPACE_ECAT6
138  // TODO get next info from OutputFileFormat class instead of hard-wiring
139  // this in here
140  const bool supports_different_xy_pixel_sizes =
141  dynamic_cast<ECAT6OutputFileFormat const * const>(_output_file_format_sptr.get()) == 0
142  ? true : false;
143  const bool supports_origin_z_shift =
144  dynamic_cast<ECAT6OutputFileFormat const * const>(_output_file_format_sptr.get()) == 0
145  ? true : false;
146  const bool supports_origin_xy_shift = true;
147 #else
148  const bool supports_different_xy_pixel_sizes = true;
149  const bool supports_origin_z_shift = true;
150  const bool supports_origin_xy_shift = true;
151 #endif
152 
153  CartesianCoordinate3D<float> origin (0.F,0.F,0.F);
154  if (supports_origin_xy_shift)
155  { origin.x()=2.4F; origin.y() = -3.5F; }
156  if (supports_origin_z_shift)
157  { origin.z()=6.4F; }
158 
159  CartesianCoordinate3D<float> grid_spacing (3.F,4.F,supports_different_xy_pixel_sizes?5.F:4.F);
160 
161  IndexRange<3>
162  range(CartesianCoordinate3D<int>(0,-15,-14),
163  CartesianCoordinate3D<int>(4,14,14));
164 
165  VoxelsOnCartesianGrid<float> single_image(range,origin, grid_spacing);
166 
167  // fill with some data
168  for (int z=single_image.get_min_z(); z<=single_image.get_max_z(); ++z)
169  for (int y=single_image.get_min_y(); y<=single_image.get_max_y(); ++y)
170  for (int x=single_image.get_min_x(); x<=single_image.get_max_x(); ++x)
171  single_image[z][y][x]=
172  3*sin(static_cast<float>(x*_PI)/single_image.get_max_x())
173  *sin(static_cast<float>(y+10*_PI)/single_image.get_max_y())
174  *cos(static_cast<float>(z*_PI/3)/single_image.get_max_z());
175 
176  shared_ptr<ExamInfo> exam_info_sptr = single_image.get_exam_info_sptr();
177  exam_info_sptr->time_frame_definitions.set_num_time_frames(1);
178  exam_info_sptr->time_frame_definitions.set_time_frame(1,10,100);
179  exam_info_sptr->start_time_in_secs_since_1970 = double(1277478034);
180  exam_info_sptr->set_high_energy_thres(100.);
181  exam_info_sptr->set_low_energy_thres(5.);
182 
183  shared_ptr<VoxelsOnCartesianGrid<float> > single_image_sptr(
184  new VoxelsOnCartesianGrid<float>(single_image));
185  return single_image_sptr;
186 }
