184 void
185 Array<1, elemT>::resize(const int min_index, const int max_index)
186 {
187  this->check_state();
188  const int oldstart = this->get_min_index();
189  const int oldlength = this->size();
190 
191  base_type::resize(min_index, max_index);
192  if (oldlength == 0)
193  {
194  for (int i=this->get_min_index(); i<=this->get_max_index(); i++)
195  this->num[i] = elemT(0);
196  }
197  else
198  {
199  for (int i=this->get_min_index(); i<oldstart && i<=this->get_max_index(); ++i)
200  this->num[i] = elemT(0);
201  for (int i=std::max(oldstart + oldlength, this->get_min_index()); i<=this->get_max_index(); ++i)
202  this->num[i] = elemT(0);
203  }
204  this->check_state();
205 }
