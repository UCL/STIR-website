87  virtual bool can_read( const FileSignature& signature, const std::string& filename) const
88  {
89  // Looking for the right key in the parameter file
90  std::ifstream par_file(filename.c_str());
91  std::string key;
92  std::getline(par_file, key, ':');
93  key = standardise_interfile_keyword(key);
94  if( key != std::string("clistmodedatasafir parameters")) {
95  return false;
96  }
97  if( !actual_do_parsing(filename) ) return false;
98  std::ifstream data_file(listmode_filename.c_str(), std::ios::binary);
99  char* buffer = new char[32];
100  data_file.read(buffer, 32);
101  bool cr = (!strncmp(buffer, "MUPET CListModeData\0", 20) || !strncmp(buffer, "SAFIR CListModeData\0", 20));
102  if( !cr ) {
103  warning("SAFIRCListModeInputFileFormat tried to read file " + listmode_filename + " but it seems to have the wrong signature.");
104  }
105 
106  delete[] buffer;
107  return cr;
108  }
