43 inline double
44 decay_correction_factor(const double isotope_halflife, const double start_time, const double end_time)
45 {
46  assert(end_time-start_time>0);
47  const double lambda=std::log(2.)/isotope_halflife;
48 
49  return
50  std::fabs(lambda*(end_time-start_time)) < .01
51  ? std::exp(-start_time*lambda) // if very short frame, we can ignore the duration
52  : lambda*(end_time-start_time)/
53  (std::exp(-start_time*lambda)-std::exp(-end_time*lambda));
54 }
