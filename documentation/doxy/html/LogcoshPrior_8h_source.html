193  static inline float logcosh(const float d)
194  {
195  const float x = fabs(d);
196  if ( x < 30.f ){
197  return log(cosh(x));
198  } else {
199  return x + log(0.5f);
200  }
201  }

210  static inline float surrogate(const float d, const float scalar)
211  {
212  const float eps = 0.01;
213  const float x = d * scalar;
214  // If abs(x) is less than eps,
215  // use Taylor approximatation of tanh: tanh(x)/x ~= (x - x^3/3)/x = 1- x^2/3.
216  // Prevents divide by zeros
217  if (fabs(x)<eps)
218  { return 1- square(x)/3; }
219  else
220  { return tanh(x)/x; }
221  }
222

230  static inline float Hessian(const float d, const float scalar)
231  {
232  const float x = d * scalar;
233  return square((1/ cosh(x)));
