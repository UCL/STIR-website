125 public:

128  explicit JacobianForIntBP(const ProjDataInfoCylindrical* proj_data_info_ptr, bool exact);
129  // s in mm here!
130  float operator()(const float delta, const float s) const
131  {
132  float tmp;
133  if (use_exact_Jacobian_now)
134  tmp = 4*(R2 - s*s);
135  else
136  tmp = 4*R2;
137  if (!arccor)
138  tmp *= sqrt(tmp);
139  return
140  (arccor ? tmp : pow(tmp,1.5F)) /
141  pow(tmp + ring_spacing2*delta*delta, 1.5F)* backprojection_normalisation;
142  }
