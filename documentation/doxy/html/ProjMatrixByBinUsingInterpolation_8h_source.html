137  // s in mm here!
138  float operator()(const float delta, const float s) const
139  {
140  float tmp;
141  if (use_exact_Jacobian_now)
142  tmp = 4*(R2 - s*s);
143  else
144  tmp = 4*R2;
145  if (!arccor)
146  tmp *= sqrt(tmp);
147  return
148  (arccor ? tmp : pow(tmp,1.5F)) /
149  pow(tmp + ring_spacing2*delta*delta, 1.5F)* backprojection_normalisation;
150  }
