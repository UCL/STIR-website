30template <class ProjDataInfoT>
31CListEventScannerWithDiscreteDetectors<ProjDataInfoT>::CListEventScannerWithDiscreteDetectors(
32 const shared_ptr<const ProjDataInfo>& proj_data_info_sptr)
33{
34 if (!proj_data_info_sptr)
35 error("CListEventScannerWithDiscreteDetectors constructor called with zero pointer");
36
37 auto scanner_sptr = proj_data_info_sptr->get_scanner_sptr();
38 // get bare pointer of uncompressed ProjDataInfo
39 auto pdi_ptr = ProjDataInfo::construct_proj_data_info(scanner_sptr,
40 1,
41 scanner_sptr->get_num_rings() - 1,
42 scanner_sptr->get_num_detectors_per_ring() / 2,
43 scanner_sptr->get_max_num_non_arccorrected_bins(),
44 /* arc correction = */ false,
45 /* TOF mashing = */ 1)
46 .release();
47 // check type
48 auto pdi_ptr_cast = dynamic_cast<ProjDataInfoT*>(pdi_ptr);
49 if (!pdi_ptr_cast)
50 {
51 delete pdi_ptr;
52 error("CListEventScannerWithDiscreteDetectors constructor called with scanner that gives wrong type of ProjDataInfo");
53 }
54 // set shared_ptr from bare pointer (will take ownership)
55 this->uncompressed_proj_data_info_sptr.reset(pdi_ptr_cast);
56}

58template <class ProjDataInfoT>
59LORAs2Points<float>
60CListEventScannerWithDiscreteDetectors<ProjDataInfoT>::get_LOR() const
61{
62 LORAs2Points<float> lor;
63 const bool swap = true;
64 // provide somewhat shorter names for the 2 coordinates, taking swap into account
65 CartesianCoordinate3D<float>& coord_1 = swap ? lor.p2() : lor.p1();
66 CartesianCoordinate3D<float>& coord_2 = swap ? lor.p1() : lor.p2();
67
68 DetectionPositionPair<> det_pos;
69 this->get_detection_position(det_pos);
70 assert(det_pos.pos1().radial_coord() == 0);
71 assert(det_pos.pos2().radial_coord() == 0);
72
73 // TODO we're using an obsolete function here which uses a different coordinate system
74 this->get_uncompressed_proj_data_info_sptr()->find_cartesian_coordinates_given_scanner_coordinates(
75 coord_1,
76 coord_2,
77 det_pos.pos1().axial_coord(),
78 det_pos.pos2().axial_coord(),
79 det_pos.pos1().tangential_coord(),
80 det_pos.pos2().tangential_coord(),
81 det_pos.timing_pos());
82 // find shift in z
83 const float shift = this->get_uncompressed_proj_data_info_sptr()->get_ring_spacing()
84 * (this->get_uncompressed_proj_data_info_sptr()->get_scanner_ptr()->get_num_rings() - 1) / 2.F;
85 coord_1.z() -= shift;
86 coord_2.z() -= shift;
87
88 return lor;
89}

91template <class ProjDataInfoT>
92void
93CListEventScannerWithDiscreteDetectors<ProjDataInfoT>::get_bin(Bin& bin, const ProjDataInfo& proj_data_info) const
94{
95 assert(dynamic_cast<ProjDataInfoT const*>(&proj_data_info) != 0);
96 DetectionPositionPair<> det_pos;
97 this->get_detection_position(det_pos);
98 if (static_cast<ProjDataInfoT const&>(proj_data_info).get_bin_for_det_pos_pair(bin, det_pos) == Succeeded::no)
99 bin.set_bin_value(0);
100 else
101 {
102 bin.set_bin_value(1);
103 }
104}
