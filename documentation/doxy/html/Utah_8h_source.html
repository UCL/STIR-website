137 shared_ptr<Shape3D>
138 Utah_phantom::make_inside_B_ptr(const float fraction) const
139 {
140  // first get a new copy of full_B
141  shared_ptr<Shape3D> small_full_B_ptr = get_full_B_ptr()->clone();
142  // make it smaller
143  small_full_B_ptr ->scale_around_origin(CartesianCoordinate3D<float>(1.F,fraction,fraction));
144 
145  // the same for D
146  shared_ptr<Shape3D> large_D_ptr = get_D_ptr()->clone();
147  large_D_ptr ->scale_around_origin(CartesianCoordinate3D<float>(1.F,1/fraction,1/fraction));
148  // and E
149  shared_ptr<Shape3D> large_E_ptr = get_E_ptr()->clone();
150  large_E_ptr ->scale_around_origin(CartesianCoordinate3D<float>(1.F,1/fraction,1/fraction));
151 
152  // combine the whole thing
153  shared_ptr<Shape3D> small_full_B_notD_ptr =
154  new CombinedShape3D< logical_and_not<bool> > ( small_full_B_ptr,large_D_ptr);
155  return
156  new CombinedShape3D< logical_and_not<bool> > ( small_full_B_notD_ptr,large_E_ptr);
157 }
