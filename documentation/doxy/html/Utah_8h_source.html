130 Utah_phantom::make_inside_B_ptr(const float fraction) const
131 {
132  // first get a new copy of full_B
133  shared_ptr<Shape3D> small_full_B_ptr = get_full_B_ptr()->clone();
134  // make it smaller
135  small_full_B_ptr ->scale_around_origin(CartesianCoordinate3D<float>(1.F,fraction,fraction));
136 
137  // the same for D
138  shared_ptr<Shape3D> large_D_ptr = get_D_ptr()->clone();
139  large_D_ptr ->scale_around_origin(CartesianCoordinate3D<float>(1.F,1/fraction,1/fraction));
140  // and E
141  shared_ptr<Shape3D> large_E_ptr = get_E_ptr()->clone();
142  large_E_ptr ->scale_around_origin(CartesianCoordinate3D<float>(1.F,1/fraction,1/fraction));
143 
144  // combine the whole thing
145  shared_ptr<Shape3D> small_full_B_notD_ptr =
146  new CombinedShape3D< logical_and_not<bool> > ( small_full_B_ptr,large_D_ptr);
147  return
148  new CombinedShape3D< logical_and_not<bool> > ( small_full_B_notD_ptr,large_E_ptr);
