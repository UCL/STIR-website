71 absolute_max_eigenvector_using_power_method(elemT& max_eigenvalue,
72  Array<1,elemT>& max_eigenvector,
73  const Array<2,elemT>& m,
74  const Array<1,elemT>& start,
75  const double tolerance = .01,
76  const unsigned long max_num_iterations = 10000UL)
77 {
78  assert(m.is_regular());
79  if (m.size()==0)
80  { max_eigenvalue=0; max_eigenvector=start; return Succeeded::yes; }
81 
82  const double tolerance_squared = square(tolerance);
83  Array<1,elemT> current = start;
84  current /= (*abs_max_element(current.begin(), current.end()));
85  unsigned long remaining_num_iterations = max_num_iterations;
86 
87  double change;
88  do
89  {
90  max_eigenvector = matrix_multiply(m, current);
91  const elemT norm_factor = *abs_max_element(max_eigenvector.begin(), max_eigenvector.end());
92  max_eigenvector /= norm_factor;
93  change = norm_squared(max_eigenvector - current);
94  current = max_eigenvector;
95  --remaining_num_iterations;
96  }
97  while (change > tolerance_squared && remaining_num_iterations!=0);
98 
99  current /= static_cast<elemT>(norm(current));
100  max_eigenvector = matrix_multiply(m, current);
101  // compute eigenvalue using Rayleigh quotient
102  max_eigenvalue =
103  static_cast<elemT>(inner_product(current,max_eigenvector) /
104  norm_squared(current));
105  max_eigenvector /= static_cast<elemT>(norm(max_eigenvector));
106 
107  return remaining_num_iterations ==0 ? Succeeded::no : Succeeded::yes;
108

111 }
