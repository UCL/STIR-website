81 void
82 ReconstructionTests<TargetT>::
83 construct_input_data()
84 {
85  Verbosity::set(1);
86  if (this->_proj_data_filename.empty())
87  {
88  // construct a small scanner and sinogram
89  shared_ptr<Scanner> scanner_sptr(new Scanner(Scanner::E953));
90  scanner_sptr->set_num_rings(5);
91  shared_ptr<ProjDataInfo> proj_data_info_sptr(
92  ProjDataInfo::ProjDataInfoCTI(scanner_sptr,
93  /*span=*/3,
94  /*max_delta=*/4,
95  /*num_views=*/128,
96  /*num_tang_poss=*/128));
97  shared_ptr<ExamInfo> exam_info_sptr(new ExamInfo);
98  exam_info_sptr->imaging_modality = ImagingModality::PT;
99  _proj_data_sptr.reset(new ProjDataInMemory (exam_info_sptr, proj_data_info_sptr));
100  std::cerr << "Will run tests with projection data with the following settings:\n"
101  << proj_data_info_sptr->parameter_info();
102  }
103  else
104  {
105  shared_ptr<ProjData> proj_data_sptr =
106  ProjData::read_from_file(this->_proj_data_filename);
107  _proj_data_sptr.reset(new ProjDataInMemory (*proj_data_sptr));
108  }
109 
110  if (this->_input_density_filename.empty())
111  {
112  CartesianCoordinate3D<float> origin (0,0,0);
113  const float zoom=.7F;
114 
115  shared_ptr<VoxelsOnCartesianGrid<float> >
116  vox_sptr(new VoxelsOnCartesianGrid<float>(this->_proj_data_sptr->get_exam_info_sptr(),
117  *this->_proj_data_sptr->get_proj_data_info_sptr(),
118  zoom,origin));
119 
120  // create very long cylinder, such that we don't have to think about origin
121  EllipsoidalCylinder cylinder(/*length_z*/1000.F,
122  /*radius_y*/100.F,
123  /*radius_x*/90.F,
124  CartesianCoordinate3D<float>(0.F,0.F,0.F));
125  cylinder.construct_volume(*vox_sptr, CartesianCoordinate3D<int>(2,2,2));
126 
127  // filter it a bit to avoid too high frequency stuff creating trouble in the comparison
128  SeparableGaussianImageFilter<float> filter;
129  filter.set_fwhms(make_coordinate(10.F,10.F,10.F));
130  filter.set_up(*vox_sptr);
131  filter.apply(*vox_sptr);
132  this->_input_density_sptr = vox_sptr;
133  }
134  else
135  {
136  shared_ptr<TargetT> aptr(read_from_file<TargetT>(this->_input_density_filename));
137  this->_input_density_sptr = aptr;
138  }
139 
140  // forward project
141  {
142  shared_ptr<ProjMatrixByBin> PM_sptr(new ProjMatrixByBinUsingRayTracing);
143  shared_ptr<ForwardProjectorByBin> fwd_proj_sptr =
144  MAKE_SHARED<ForwardProjectorByBinUsingProjMatrixByBin>(PM_sptr);
145  fwd_proj_sptr->set_up(this->_proj_data_sptr->get_proj_data_info_sptr(),
146  this->_input_density_sptr);
147  fwd_proj_sptr->set_input(*this->_input_density_sptr);
148  fwd_proj_sptr->forward_project(*this->_proj_data_sptr);
149  }

172 template <class TargetT>
173 void
174 ReconstructionTests<TargetT>::
175 compare(const shared_ptr<const TargetT> output_sptr)
176 {
177  if (!check(this->_input_density_sptr->has_same_characteristics(*output_sptr),
178  "output image has wrong characteristics"))
179  return;
180 
181  shared_ptr<TargetT> diff_sptr(output_sptr->clone());
182  *diff_sptr -= *this->_input_density_sptr;
183  const float diff_min = diff_sptr->find_min();
184  const float diff_max = diff_sptr->find_max();
185  const float max_input = this->_input_density_sptr->find_max();
186  in_place_abs(*diff_sptr);
187  const float mean_abs_error=diff_sptr->sum() / this->_input_density_sptr->size_all();
188  std::cerr << "Reconstruction diff relative range: "
189  << "[" << diff_min/max_input << ", " << diff_max/max_input << "]\n"
190  << "mean abs diff normalised was " << mean_abs_error/max_input << "\n";
191  if (!check_if_less(-0.3F, diff_min/max_input, "relative diff min") ||
192  !check_if_less(diff_max/max_input, .3F, "relative diff max") ||
193  !check_if_less(mean_abs_error/max_input, .01F, "relative mean abs diff"))
194  {
195  const std::string prefix = "test_recon_" + this->_recon_sptr->method_info();
196  write_to_file(prefix + "_output.hv", *output_sptr);
197  write_to_file(prefix + "_original.hv", *this->_input_density_sptr);
198  write_to_file(prefix + "_diff.hv", *diff_sptr);
199  }
200 }
