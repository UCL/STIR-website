187  inline void HighResWallClockTimer::stop(void)
188  {
189  assert(m_bRunning);
190 
191 #if defined(_AIX)
192 
193  read_real_time(&m_Finish, TIMEBASE_SZ);
194 
195  time_base_to_time(&m_Start, TIMEBASE_SZ);
196  time_base_to_time(&m_Finish, TIMEBASE_SZ);
197 
198  int Secs = m_Finish.tb_high - m_Start.tb_high;
199  int Nanosecs = m_Finish.tb_low - m_Start.tb_low;
200 
201  // If there was a carry from low-order to high-order during
202  // the measurement, we may have to undo it.
203 
204  if (Nanosecs < 0)
205  {
206  Secs--;
207  Nanosecs += 1000000000;
208  };
209 
210  m_Secs += Secs;
211  m_Nanosecs += Nanosecs;
212  if (m_Nanosecs >= 1000000000)
213  {
214  m_Secs++;
215  m_Nanosecs -= 1000000000;
216  };
217 
218 #elif defined(__sun)
219 
220  m_Finish = gethrtime();
221 
222  m_Secs += (m_Finish - m_Start) / 1000000000;
223  m_Nanosecs += (m_Finish - m_Start) % 1000000000;
224  if (m_Nanosecs >= 1000000000)
225  {
226  m_Secs++;
227  m_Nanosecs -= 1000000000;
228  };
229 
230 #elif defined(WIN32)
231 
232  BOOL Result = QueryPerformanceCounter(&m_Finish);
233  assert(Result); // if failed, high-resolution timers are not supported by this hardware
234 
235  LONGLONG Delta = m_Finish.QuadPart - m_Start.QuadPart;
236 
237  int Resolution = get_resolution_in_nanosecs();
238 
239  m_Secs += static_cast<int>(Delta / Resolution);
240  m_Nanosecs += static_cast<int>(Delta % Resolution);
241  if (m_Nanosecs >= 1000000000)
242  {
243  m_Secs++;
244  m_Nanosecs -= 1000000000;
245  }
246 
247 #elif defined(STIR_HRWCT_Use_gettimeofday)
248 
249 #ifndef NDEBUG
250  int Result =
251 #endif
252  gettimeofday(&m_Finish, NULL);
253  assert(Result == 0);
254  m_Secs += (m_Finish.tv_sec - m_Start.tv_sec);
255  int Microsecs = (m_Finish.tv_usec - m_Start.tv_usec);
256  if (Microsecs < 0)
257  {
258  m_Secs--;
259  Microsecs += 1000000;
260  };
261 
262  m_Nanosecs += (Microsecs * 1000);
263  if (m_Nanosecs >= 1000000000)
264  {
265  m_Secs++;
266  m_Nanosecs -= 1000000000;
267  };
268 
269 #endif
270 
271  assert(m_Secs >= 0);
272  assert(m_Nanosecs >= 0 && m_Nanosecs < 1000000000);
273 
274  m_bRunning = false;
275  };

328  inline int HighResWallClockTimer::get_resolution_in_nanosecs(void)
329  {
330 
331  /*static*/ int Resolution = -1; // cached resolution
332 
333  if (Resolution > 0) return Resolution;
334 
335 #if defined(_AIX)
336 
337  timebasestruct_t Start;
338  timebasestruct_t Current;
339 
340  read_real_time(&Start, TIMEBASE_SZ);
341  time_base_to_time(&Start, TIMEBASE_SZ);
342 
343  int MinSecs = -1; // no minimum yet
344  int MinNanosecs = -1;
345  int SecondsElapsed = 0;
346 
347  do
348  {
349  timebasestruct_t Times[2];
350 
351  read_real_time(&Times[0], TIMEBASE_SZ);
352  do
353  {
354  read_real_time(&Times[1], TIMEBASE_SZ);
355  assert(Times[1].flag == Times[0].flag);
356  }
357  while (Times[1].tb_high == Times[0].tb_high &&
358  Times[1].tb_low == Times[0].tb_low);
359 
360  time_base_to_time(&Times[0], TIMEBASE_SZ);
361  time_base_to_time(&Times[1], TIMEBASE_SZ);
362 
363  int Secs = Times[1].tb_high - Times[0].tb_high;
364  int Nanosecs = Times[1].tb_low - Times[0].tb_low;
365 
366  // If there was a carry from low-order to high-order during
367  // the measurement, we may have to undo it.
368 
369  if (Nanosecs < 0)
370  {
371  Secs--;
372  Nanosecs += 1000000000;
373  };
374 
375  assert(Secs >= 0 && Nanosecs >= 0 && Nanosecs < 1000000000);
376  assert(Secs != 0 || Nanosecs != 0); // shouldn't be the same as we checked in the 'do' loop
377 
378  // cout << "[" << Secs << " " << Nanosecs << "]" << endl;
379 
380  if (MinSecs < 0 || Secs < MinSecs || (Secs == MinSecs && Nanosecs < MinNanosecs))
381  {
382  MinSecs = Secs;
383  MinNanosecs = Nanosecs;
384  };
385 
386  read_real_time(&Current, TIMEBASE_SZ);
387  time_base_to_time(&Current, TIMEBASE_SZ);
388 
389  SecondsElapsed = Current.tb_high - Start.tb_high;
390  if (Current.tb_low < Start.tb_low) SecondsElapsed--;
391  }
392  while (SecondsElapsed < 1);
393 
394  assert(MinSecs < 2); // otherwise it's too coarse for this measurement
395 
396  Resolution = MinSecs * 1000000000 + MinNanosecs;
397 
398 #elif defined(__sun)
399 
400  hrtime_t Start = gethrtime();
401  hrtime_t Current;
402 
403  hrtime_t MinNanosecs = -1;
404 
405  do
406  {
407  hrtime_t Times[2];
408 
409  Times[0] = gethrtime();
410 
411  while ((Times[1] = gethrtime()) == Times[0]) /* do nothing */;
412 
413  hrtime_t Nanosecs = Times[1] - Times[0];
414 
415  assert(Nanosecs > 0);
416 
417  if (MinNanosecs < 0 || Nanosecs < MinNanosecs)
418  {
419  MinNanosecs = Nanosecs;
420  };
421 
422  Current = gethrtime();
423  }
424  while (Current - Start < 1000000000); // 1 second
425 
426  assert(MinNanosecs < 2000000000); // otherwise it's too coarse for this measurement
427 
428  Resolution = static_cast<int>(MinNanosecs);
429 
430 #elif defined(WIN32)
431 
432  LARGE_INTEGER Freq;
433  BOOL Result;
434 
435  Result = QueryPerformanceFrequency(&Freq);
436  assert(Result); // if failed, high-resolution timers are not supported by this hardware
437 
438  Resolution = static_cast<int>(1000000000 / Freq.QuadPart);
439 
440 #elif defined(STIR_HRWCT_Use_gettimeofday)
441 
442  //TODO
443 
444 #endif
445 
446  assert(Resolution > 0);
447 
448  return Resolution;
449
