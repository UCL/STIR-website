55 class JacobianForIntBP
56 {
57 private:
58  // store some scanner related data to avoid recomputation
59  const float R2;
60  const float dxy2;
61  const float ring_spacing2;
62  // total normalisation of backprojection, 3 factors:
63  // (_Pi/scanner.num_views) for discretisation of integral over phi
64  // scanner.ring_spacing for discretisation of integral over delta
65  // normalisation of projection space integral: 1/(2 Pi)
66 
67  const float backprojection_normalisation;
68 
69  const bool use_exact_Jacobian_now;
70 
71 public:
72  explicit JacobianForIntBP(const shared_ptr<const ProjDataInfoCylindricalArcCorr> proj_data_info_sptr, bool exact);
73 
74  float operator()(const float delta, const float s) const
75  {
76  float tmp;
77  if (use_exact_Jacobian_now)
78  tmp = 4*(R2 - dxy2 * s*s);
79  else
80  tmp = 4*R2;
81  return tmp / pow(tmp + ring_spacing2*delta*delta, 1.5F)* backprojection_normalisation;
82  }
83 };
