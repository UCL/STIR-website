143 standardise(const Bin& in_bin, const ProjDataInfo& proj_data_info)
144 {
145  Bin bin = in_bin;
146  bool swap_direction = false;

153  {
154  swap_direction = true;

158  assert(bin.view_num()<=proj_data_info.get_max_view_num());
159 
160  if (swap_direction)

163  bin.segment_num() *= -1;
164  }

172 {
173  LORInAxialAndNoArcCorrSinogramCoordinates<float> tmp_lor;
174  LORInAxialAndNoArcCorrSinogramCoordinates<float> lor;

183  return Succeeded::no;

186  }
