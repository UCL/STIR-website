60 static_cast<Derived*>(this)->get_data().set_sinogram_and_ring_coordinates(

62 }
63 if (this->get_uncompressed_proj_data_info_sptr()->is_tof_data())

68static void
69sinogram_coordinates_to_bin(Bin& bin,
70 const int view_num,
71 const int tang_pos_num,
72 const int ring_a,
73 const int ring_b,
74 const ProjDataInfoCylindrical& proj_data_info)
75{
76 if (proj_data_info.get_segment_axial_pos_num_for_ring_pair(bin.segment_num(), bin.axial_pos_num(), ring_a, ring_b)
77 == Succeeded::no)
78 {
79 bin.set_bin_value(-1);
80 return;
81 }
82 bin.set_bin_value(1);
83 bin.view_num() = view_num / proj_data_info.get_view_mashing_factor();
84 bin.tangential_pos_num() = tang_pos_num;
