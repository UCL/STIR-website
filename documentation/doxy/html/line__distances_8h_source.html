47 template <class coordT>
48 inline
49 coordT
50 coordinate_between_2_lines(CartesianCoordinate3D<coordT>& result,
51  const LORAs2Points<coordT>& line0,
52  const LORAs2Points<coordT>& line1)
53 {
54  /* Rationale:
55  parametrise points on the lines as
56  r0 + a0 d0, r1 + a1 d1
57  where r0 is a point on the line, and d0 is its direction
58 
59  compute distance squared between those points, i.e.
60  norm(a0 d0 - a1 d1 + r0-r1)^2
61  This can be written using inner products of all the vectors.
62 
63  Minimising it (by computing derivatives) results in a linear eq in a0, a1:
64  a0 d0d0 == a1 d0d1 + r10d0,
65  a0 d0d1 == a1 d1d1 + r10d1
66  which is easily solved.
67  The half-way point can then be found by using
68  (r0 + a0 d0 + r1 + a1 d1)/2
69  for the a0,a1 found.
70  */
71  const CartesianCoordinate3D<coordT>& r0 = line0.p1();
72  const CartesianCoordinate3D<coordT>& r1 = line1.p1();
73  const CartesianCoordinate3D<coordT> r10 = r1-r0;
74  const CartesianCoordinate3D<coordT> d0 = line0.p2() - line0.p1();
75  const CartesianCoordinate3D<coordT> d1 = line1.p2() - line1.p1();
76  const coordT d0d0 = inner_product(d0,d0);
77  const coordT d0d1 = inner_product(d0,d1);
78  const coordT d1d1 = inner_product(d1,d1);
79  const coordT r10d0 = inner_product(r10,d0);
80  const coordT r10r10 = inner_product(r10,r10);
81 
82  const coordT eps=d0d0*10E-5; // small number for comparisons
83 
84  const coordT denom = square(d0d1) - d0d0*d1d1;
85  coordT distance_squared;
86  if (std::fabs(denom) <= eps)
87  {
88  //parallel lines
89  const coordT a0 = r10d0/d0d0;
90  result = r0 + d0*a0;
91  distance_squared = r10r10 - square(r10d0)/d0d0;
92  }
93  else
94  {
95  const coordT r10d1 = inner_product(r10,d1);
96  const coordT a0 = (-d1d1*r10d0 + d0d1*r10d1)/denom;
97  const coordT a1 = (-d0d1*r10d0 + d0d0*r10d1)/denom;
98 
99  result = ((r0 + d0*a0) + (r1 + d1*a1))/2;
100  distance_squared =
101  (d1d1*square(r10d0) - 2*d0d1*r10d0*r10d1 + d0d0*square(r10d1))/denom +
102  r10r10;
103  }
104  if (distance_squared >= 0)
105  return std::sqrt(distance_squared);
106  else
107  {
108  if (-distance_squared < eps)
109  return 0;
110  else
111  {
112  assert(false);
113  return std::sqrt(distance_squared); // will return NaN
114  }
115  }
116 }
